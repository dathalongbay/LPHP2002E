            margin-top: 50px;
            margin-bottom: 40px;
            margin-left: 10px;
            margin-right: 10px;
        }

        => viết tắt
        .classA {
            margin: A B C;
        }
        trong đó A là margin top
        B vừa là margin left và cũng margin right
        C là margin bottom
